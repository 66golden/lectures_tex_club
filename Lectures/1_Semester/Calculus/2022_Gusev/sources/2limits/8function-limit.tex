\begin{theorem}
	Если $\fnis{f, g, h} D \mapsto \R$, $x_0 \in \RR$, $\forall \delta > 0\ \UU_\delta(x_0) \cap D \ne \varnothing$, $f(x) \le h(x) \le g(x)$ и $\lim_{x \to x_0} f(x) = \lim_{x \to x_0} g(x) = a$, то $\lim_{x \to x_0} h(x) = a$.
\end{theorem}
